```typescript
"use client";
import { useCallback, useEffect, useState } from "react";
import { createClient } from "@/lib/supabase/client";
import type { Database } from "@/lib/supabase/types";

type DashboardRow = Database["public"]["Views"]["v_orders_dashboard"]["Row"];

export type OrderStatus = "pending" | "approved" | "rejected" | "all";

export interface UseOrdersOptions {
  status: OrderStatus;
  erpSystems: string[]; // 예: ['gl','gl_pharm','hnb']
  txTypes: string[]; // 예: ['purchase','sale','return_sale','return_purchase']
  itemSearch: string; // 품목명 또는 erp_code 통합 검색
  dateFrom: string | null; // 'YYYY-MM-DD'
  dateTo: string | null;
  page: number; // 0-base
  pageSize: number; // 기본 50
}

export interface UseOrdersResult {
  rows: DashboardRow[];
  totalCount: number;
  loading: boolean;
  error: string | null;
  refetch: () => void;
}

export function useOrders(opts: UseOrdersOptions): UseOrdersResult {
  const [rows, setRows] = useState<DashboardRow[]>([]);
  const [totalCount, setTotalCount] = useState(0);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const supabase = createClient();

  const fetchRows = useCallback(async () => {
    setLoading(true);
    setError(null);
    let q = supabase.from("v_orders_dashboard").select("*", { count: "exact" });

    if (opts.status !== "all") {
      q = q.eq("status", opts.status);
    }
    if (opts.erpSystems.length > 0) {
      q = q.in("erp_system", opts.erpSystems);
    }
    if (opts.txTypes.length > 0) {
      q = q.in("tx_type", opts.txTypes);
    }
    if (opts.itemSearch.trim()) {
      const s = opts.itemSearch.trim();
      q = q.or(`item_name.ilike.%${s}%,erp_code.ilike.%${s}%`);
    }
    if (opts.dateFrom) q = q.gte("tx_date", opts.dateFrom);
    if (opts.dateTo) q = q.lte("tx_date", opts.dateTo);

    const from = opts.page * opts.pageSize;
    const to = from + opts.pageSize - 1;
    q = q
      .order("tx_date", { ascending: false })
      .order("order_id", { ascending: false })
      .range(from, to);

    const { data, count, error: err } = await q;
    if (err) {
      setError(err.message);
      setRows([]);
      setTotalCount(0);
    } else {
      setRows(data ?? []);
      setTotalCount(count ?? 0);
    }
    setLoading(false);
  }, [
    supabase,
    opts.status,
    opts.erpSystems,
    opts.txTypes,
    opts.itemSearch,
    opts.dateFrom,
    opts.dateTo,
    opts.page,
    opts.pageSize,
  ]);

  useEffect(() => {
    void fetchRows();
  }, [fetchRows]);

  return { rows, totalCount, loading, error, refetch: fetchRows };
}
```

**`useStockMovementsInboundReturn.ts`:**

```typescript
"use client";
import { useEffect, useState } from "react";
import { createClient } from "@/lib/supabase/client";
import type { Database } from "@/lib/supabase/types";

type Movement = Database["public"]["Tables"]["stock_movement"]["Row"];

export function useStockMovementsInboundReturn(itemId: number | null) {
  const [rows, setRows] = useState<Movement[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const supabase = createClient();

  useEffect(() => {
    if (itemId === null) {
      setRows([]);
      return;
    }
    let cancelled = false;
    setLoading(true);
    setError(null);

    void (async () => {
      const { data, error: err } = await supabase
        .from("stock_movement")
        .select("*")
        .eq("item_id", itemId)
        .in("movement_type", ["purchase", "return_sale"])
        .order("movement_date", { ascending: false })
        .order("id", { ascending: false });

      if (cancelled) return;
      if (err) {
        setError(err.message);
        setRows([]);
      } else {
        setRows(data ?? []);
      }
      setLoading(false);
    })();

    return () => {
      cancelled = true;
```

```typescript
// src/app/api/orders/approve/route.ts
// 승인: orders.status = 'approved' (UPDATE 트리거가 stock_movement 자동 생성)
// 승인 취소: orders.status = 'pending' (트리거가 stock_movement 자동 삭제)
// 요청: { orderIds: number[], action: 'approve' | 'unapprove' }

import { createClient } from "@/lib/supabase/server";
import { createClient as createAdmin } from "@supabase/supabase-js";
import { NextResponse } from "next/server";
import type { Database } from "@/lib/supabase/types";

interface Body {
  orderIds?: number[];
  action?: "approve" | "unapprove";
}

function parseBody(raw: unknown): { orderIds: number[]; action: "approve" | "unapprove" } | null {
  if (raw === null || typeof raw !== "object") return null;
  const o = raw as Body;
  if (!Array.isArray(o.orderIds) || o.orderIds.length === 0) return null;
  const orderIds = o.orderIds.filter((n): n is number => Number.isInteger(n) && n > 0);
  if (orderIds.length === 0) return null;
  if (o.action !== "approve" && o.action !== "unapprove") return null;
  return { orderIds, action: o.action };
}

export async function POST(request: Request) {
  // 인증
  const supabase = await createClient();
  const {
    data: { user },
  } = await supabase.auth.getUser();
  if (!user) {
    return NextResponse.json({ error: "인증이 필요합니다." }, { status: 401 });
  }

  // 환경변수
  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const serviceRoleKey = process.env.SUPABASE_SERVICE_ROLE_KEY;
  if (!supabaseUrl || !serviceRoleKey) {
    return NextResponse.json({ error: "Supabase 서버 환경변수 미설정" }, { status: 500 });
  }

  // 요청 검증
  let json: unknown;
  try {
    json = await request.json();
  } catch {
    return NextResponse.json({ error: "JSON 본문을 읽을 수 없습니다." }, { status: 400 });
  }
  const parsed = parseBody(json);
  if (!parsed) {
    return NextResponse.json(
      { error: "orderIds(양의 정수 배열) + action('approve'|'unapprove') 필요" },
      { status: 400 }
    );
  }

  // 업데이트 (트리거가 stock_movement 반영)
  const admin = createAdmin<Database>(supabaseUrl, serviceRoleKey);
  const newStatus = parsed.action === "approve" ? "approved" : "pending";
  const payload: Record<string, unknown> = { status: newStatus };
  if (parsed.action === "approve") {
    payload.approved_by = user.email ?? user.id;
    payload.approved_at = new Date().toISOString();
  } else {
    payload.approved_by = null;
    payload.approved_at = null;
  }

  const { data, error } = await admin
    .from("orders")
    .update(payload)
    .in("id", parsed.orderIds)
    .select("id, status");

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  return NextResponse.json({
    message: parsed.action === "approve" ? "승인 완료" : "승인 취소 완료",
    updated: data?.length ?? 0,
    ids: data?.map((r) => r.id) ?? [],
  });
```

```typescript
      return NextResponse.json({ error: `${i + 1}행: 날짜 형식 오류` }, { status: 400 });
    if (!erpCode) return NextResponse.json({ error: `${i + 1}행: 품목코드 누락` }, { status: 400 });
    if (quantity === null || quantity <= 0)
      return NextResponse.json({ error: `${i + 1}행: 수량 오류` }, { status: 400 });
    if (unitPrice === null || unitPrice < 0)
      return NextResponse.json({ error: `${i + 1}행: 단가 오류` }, { status: 400 });
    if (total === null || total < 0)
      return NextResponse.json({ error: `${i + 1}행: 합계 오류` }, { status: 400 });
    const expected = round2(quantity * unitPrice);
    if (Math.abs(expected - total) > 0.02) {
      return NextResponse.json(
        { error: `${i + 1}행: 합계(${total}) ≠ 수량×단가(${expected})` },
        { status: 400 }
      );
    }
    if (!supplierName)
      return NextResponse.json({ error: `${i + 1}행: 거래처 누락` }, { status: 400 });
    normalized.push({
      erpRef,
      purchaseDateIso: dateIso,
      erpCode,
      productName:
        typeof r.productName === "string" ? r.productName.trim() || undefined : undefined,
      quantity,
      unitPriceCny: round2(unitPrice),
      totalCny: round2(total),
      supplierName,
    });
  }

  const admin = createAdmin<Database>(supabaseUrl, serviceRoleKey);

  // 중복 체크: 같은 erp_system + tx_type + erp_tx_no + erp_tx_line_no UNIQUE
  // (v6의 fix_orders_erp_tx_unique_non_partial 마이그으로 전체 테이블 UNIQUE)
  // 엑셀 대량 업로드는 erp_tx_line_no를 1부터 순번 할당
  const refs = [...new Set(normalized.map((r) => r.erpRef))];
  const existing = new Set<string>();
  for (const batch of chunk(refs, 150)) {
    const { data, error } = await admin
      .from("orders")
      .select("erp_tx_no")
      .eq("erp_system", companyCode)
      .eq("tx_type", "purchase")
      .in("erp_tx_no", batch);
    if (error) {
      return NextResponse.json({ error: `중복 체크 실패: ${error.message}` }, { status: 500 });
    }
    for (const row of data ?? []) {
      if (row.erp_tx_no) existing.add(row.erp_tx_no);
    }
  }

  const toInsert = normalized.filter((r) => !existing.has(r.erpRef));
  const skipped = normalized.length - toInsert.length;

  // item_id 매핑 (item_erp_mapping으로 erp_code → item_id)
  const codes = [...new Set(toInsert.map((r) => r.erpCode))];
  const itemByCode = new Map<string, number>();
  for (const batch of chunk(codes, 150)) {
    const { data, error } = await admin
      .from("item_erp_mapping")
      .select("item_id, erp_code")
      .eq("erp_system", companyCode)
      .eq("mapping_status", "verified")
      .in("erp_code", batch);
    if (error) {
      return NextResponse.json({ error: `품목 매핑 조회 실패: ${error.message}` }, { status: 500 });
    }
    for (const m of data ?? []) {
      if (m.erp_code) itemByCode.set(m.erp_code, m.item_id);
    }
  }

  // 매핑 안 되는 행 건너뜀 + 이력에 skipped 증가
  const unmapped: string[] = [];
  const inserts: OrderInsert[] = [];
  for (let idx = 0; idx < toInsert.length; idx += 1) {
    const r = toInsert[idx];
    const itemId = itemByCode.get(r.erpCode);
    if (itemId === undefined) {
      unmapped.push(r.erpRef);
      continue;
    }
    inserts.push({
      tx_date: r.purchaseDateIso,
      item_id: itemId,
      erp_system: companyCode,
      tx_type: "purchase",
      erp_code: r.erpCode,
      erp_tx_no: r.erpRef,
      erp_tx_line_no: idx + 1,
      erp_item_name_raw: r.productName ?? null,
      counterparty: r.supplierName,
      // is_internal: DB 트리거가 자동 설정
      quantity: r.quantity,
      unit_price: r.unitPriceCny,
      total_amount: r.totalCny,
      memo: `[엑셀업로드:${companyCode}]${fileName ? " " + fileName : ""}`,
      status: "pending",
    });
  }
  const unmappedCount = unmapped.length;

  if (inserts.length > 0) {
    const { error: insertErr } = await admin.from("orders").insert(inserts);
    if (insertErr) {
      return NextResponse.json(
        { error: `orders 저장 실패: ${insertErr.message}` },
        { status: 500 }
      );
    }
  }

  // 업로드 이력
  if (fileName) {
    const { error: logErr } = await admin.from("order_excel_upload_logs").insert({
      company_code: companyCode,
      file_name: fileName,
      total_input: normalized.length,
      inserted_count: inserts.length,
      skipped_count: skipped + unmappedCount,
    });
    if (logErr) {
      return NextResponse.json({ error: `이력 저장 실패: ${logErr.message}` }, { status: 500 });
    }
  }

  return NextResponse.json({
    message: "가져오기 완료",
    totalInput: normalized.length,
    inserted: inserts.length,
    skipped,
    unmapped: unmappedCount,
    unmappedRefs: unmapped,
  });
}
```

```typescript
// src/app/api/orders/manual-erp-purchase/route.ts
// 수동 1건 입력 → orders INSERT (status='pending', tx_type='purchase')
import { createClient } from "@/lib/supabase/server";
import { createClient as createAdmin } from "@supabase/supabase-js";
import { NextResponse } from "next/server";
import type { Database } from "@/lib/supabase/types";

interface Body {
  companyCode?: "gl" | "gl_pharm" | "hnb";
  itemId?: number;
  purchaseDate?: string; // YYYY-MM-DD
  quantity?: number;
  unitPrice?: number;
  grossTotal?: number;
  supplierName?: string;
  memo?: string;
}

export async function POST(request: Request) {
  const supabase = await createClient();
  const {
    data: { user },
  } = await supabase.auth.getUser();
  if (!user) {
    return NextResponse.json({ error: "인증 필요" }, { status: 401 });
  }

  const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL;
  const serviceRoleKey = process.env.SUPABASE_SERVICE_ROLE_KEY;
  if (!supabaseUrl || !serviceRoleKey) {
    return NextResponse.json({ error: "환경변수 미설정" }, { status: 500 });
  }

  let body: Body;
  try {
    body = await request.json();
  } catch {
    return NextResponse.json({ error: "JSON 파싱 실패" }, { status: 400 });
  }

  if (!body.companyCode || !["gl", "gl_pharm", "hnb"].includes(body.companyCode)) {
    return NextResponse.json({ error: "companyCode 오류" }, { status: 400 });
  }
  if (!Number.isInteger(body.itemId) || (body.itemId ?? 0) <= 0) {
    return NextResponse.json({ error: "itemId(bigint) 필요" }, { status: 400 });
  }
  if (!body.purchaseDate || !/^\d{4}-\d{2}-\d{2}$/.test(body.purchaseDate)) {
    return NextResponse.json({ error: "purchaseDate 형식 오류" }, { status: 400 });
  }
  if (!Number.isInteger(body.quantity) || (body.quantity ?? 0) <= 0) {
    return NextResponse.json({ error: "quantity 필요" }, { status: 400 });
  }
  if (typeof body.unitPrice !== "number" || body.unitPrice < 0) {
    return NextResponse.json({ error: "unitPrice 필요" }, { status: 400 });
  }
  if (typeof body.grossTotal !== "number" || body.grossTotal < 0) {
    return NextResponse.json({ error: "grossTotal 필요" }, { status: 400 });
  }
  if (!body.supplierName || !body.supplierName.trim()) {
    return NextResponse.json({ error: "거래처 필요" }, { status: 400 });
  }

  const admin = createAdmin<Database>(supabaseUrl, serviceRoleKey);
  const { data, error } = await admin
    .from("orders")
    .insert({
      tx_date: body.purchaseDate,
      item_id: body.itemId!,
      erp_system: body.companyCode,
      tx_type: "purchase",
      erp_tx_no: `MANUAL-${Date.now()}`, // 수동 입력 고유 키
      erp_tx_line_no: 1,
      counterparty: body.supplierName.trim(),
      quantity: body.quantity!,
      unit_price: body.unitPrice,
      total_amount: body.grossTotal,
      memo: `[수동입력:${body.companyCode}]${body.memo ? " " + body.memo : ""}`,
      status: "pending",
    })
    .select("id")
    .single();

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  return NextResponse.json({ message: "저장 완료", orderId: data?.id });
}
```

- **지엘팜 "일자-No." 포맷**: 현재 구현은 `rsplit("-", n=1)` 로 뒷자리 분리. 실제 포맷이 `2024/01/15-1` 같은 슬래시 포맷이면 `doc_date` 파싱이 실패해 행 드롭될 수 있음. 실행 후 `gl_pharm` 결과 0행이면 이 부분 의심.
- **병렬 3기업 동시 로그인**: 서버 rate limit 트리거 가능성 있음. 증상 시 `crawl_all_companies` 에서 `asyncio.gather` → 순차 + 기업간 3초 딜레이로 fallback.
